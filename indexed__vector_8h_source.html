37 template <class T>
38 class IndexedVector : public Vector<T> {
39  ordered_map<cstring, const IDeclaration *> declarations;
40  bool invalid = false; // set when an error occurs; then we don't
41  // expect the validity check to succeed.
42 
43  void insertInMap(const T *a) {
44  if (a == nullptr || !a->template is<IDeclaration>()) return;
45  auto decl = a->template to<IDeclaration>();
46  auto name = decl->getName().name;
47  auto previous = declarations.find(name);
48  if (previous != declarations.end()) {
49  invalid = true;
50  ::error(ErrorType::ERR_DUPLICATE, "%1%: Duplicates declaration %2%", a,
51  previous->second);
52  } else {
53  declarations[name] = decl;
54  }
55  }
56  void removeFromMap(const T *a) {
57  if (a == nullptr) return;
58  auto decl = a->template to<IDeclaration>();
59  if (decl == nullptr) return;
60  cstring name = decl->getName().name;
61  auto it = declarations.find(name);
62  if (it == declarations.end()) BUG("%1% does not exist", a);
63  declarations.erase(it);
64  }
65 
66  public:
67  using Vector<T>::begin;
68  using Vector<T>::end;
69 
70  IndexedVector() = default;
71  IndexedVector(const IndexedVector &) = default;
72  IndexedVector(IndexedVector &&) = default;
73  IndexedVector(const std::initializer_list<const T *> &a) : Vector<T>(a) {
74  for (auto el : *this) insertInMap(el);
75  }
76  IndexedVector &operator=(const IndexedVector &) = default;
77  IndexedVector &operator=(IndexedVector &&) = default;
78  explicit IndexedVector(const T *a) { push_back(std::move(a)); }
79  explicit IndexedVector(const safe_vector<const T *> &a) {
80  insert(Vector<T>::end(), a.begin(), a.end());
81  }
82  explicit IndexedVector(const Vector<T> &a) { insert(Vector<T>::end(), a.begin(), a.end()); }
83  explicit IndexedVector(JSONLoader &json);

85  void clear() {
86  IR::Vector<T>::clear();
87  declarations.clear();
88  }

94  const IDeclaration *getDeclaration(cstring name) const {
95  auto it = declarations.find(name);
96  if (it == declarations.end()) return nullptr;
97  return it->second;
98  }
99  template <class U>
100  const U *getDeclaration(cstring name) const {
101  auto it = declarations.find(name);
102  if (it == declarations.end()) return nullptr;
103  return it->second->template to<U>();
104  }
105  Util::Enumerator<const IDeclaration *> *getDeclarations() const {
106  return Util::enumerate(Values(declarations));
107  }
108  iterator erase(iterator i) {
109  removeFromMap(*i);
110  return Vector<T>::erase(i);
111  }
112  template <typename ForwardIter>
113  iterator insert(iterator i, ForwardIter b, ForwardIter e) {
114  for (auto it = b; it != e; ++it) insertInMap(*it);
115  return Vector<T>::insert(i, b, e);
116  }
117  iterator replace(iterator i, const T *v) {
118  removeFromMap(*i);
119  *i = v;
120  insertInMap(v);
121  return ++i;
122  }
123  template <typename Container>
124  iterator append(const Container &toAppend) {
125  return insert(Vector<T>::end(), toAppend.begin(), toAppend.end());
126  }
127  template <typename Container>
128  iterator prepend(const Container &toAppend) {
129  return insert(Vector<T>::begin(), toAppend.begin(), toAppend.end());
130  }
131  iterator insert(iterator i, const T *v) {
132  insertInMap(v);
133  return Vector<T>::insert(i, v);
134  }
135  template <class... Args>
136  void emplace_back(Args &&...args) {
137  auto el = new T(std::forward<Args>(args)...);
138  push_back(el);
139  }
140  bool removeByName(cstring name) {
141  for (auto it = begin(); it != end(); ++it) {
142  auto decl = (*it)->template to<IDeclaration>();
143  if (decl != nullptr && decl->getName() == name) {
144  erase(it);
145  return true;
146  }
147  }
148  return false;
149  }

151  CHECK_NULL(a);
152  Vector<T>::push_back(a);

154  }

161  if (Vector<T>::empty()) BUG("pop_back from empty IndexedVector");
162  auto last = Vector<T>::back();
163  removeFromMap(last);
164  Vector<T>::pop_back();
165  }

182  * that does not appear anywhere in any .def file, which won't work.
183  * To make double-dispatch comparisons work, the IR generator must know
184  * about ALL instantiations of IR class templates, which it does by scanning
185  * all the .def files for instantiations. This could in theory be fixed by

189  * To avoid this problem, you need to have your code ONLY use instantiations
