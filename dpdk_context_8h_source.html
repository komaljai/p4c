84 struct TopLevelCtxt {
85  cstring progName;
86  cstring buildDate;
87  cstring compileCommand;
88  cstring compilerVersion;
89  void initTopLevelCtxt(DpdkOptions &options) {
90  buildDate = options.getBuildDate();
91  compileCommand = options.getCompileCommand();
92  progName = options.file;
93  auto fileName = progName.findlast('/');
94  // Handle the case when input file is in the current working directory.
95  // fileName would be null in that case, hence progName should remain unchanged.
96  if (fileName) progName = cstring(fileName);
97  auto fileext = progName.find(".");
98  progName = progName.replace(cstring(fileext), cstring::empty);
99  progName = progName.trim("/\t\n\r");
100  compilerVersion = options.compilerVersion;
101  }
102 };

105 struct SelectionTable {
106  unsigned max_n_groups;
107  unsigned max_n_members_per_group;
108  unsigned bound_to_action_data_table_handle;
109  void setAttributes(const IR::P4Table *tbl,
110  const std::map<const cstring, struct TableAttributes> &tableAttrmap) {
111  max_n_groups = 0;
112  max_n_members_per_group = 0;
113  auto n_groups = tbl->properties->getProperty("n_groups_max"_cs);
114  if (n_groups) {
115  auto n_groups_expr = n_groups->value->to<IR::ExpressionValue>()->expression;
116  max_n_groups = n_groups_expr->to<IR::Constant>()->asInt();
117  }
118  auto n_members = tbl->properties->getProperty("n_members_per_group_max"_cs);
119  if (n_members) {
120  auto n_members_expr = n_members->value->to<IR::ExpressionValue>()->expression;
121  max_n_members_per_group = n_members_expr->to<IR::Constant>()->asInt();
122  }
123  // Fetch associated member table handle
124  cstring actionDataTableName = tbl->name.originalName;
125  actionDataTableName = actionDataTableName.replace("_sel"_cs, cstring::empty);
126  auto actionTableAttr = ::get(tableAttrmap, actionDataTableName);
127  bound_to_action_data_table_handle = actionTableAttr.tableHandle;
128  }
129 };
