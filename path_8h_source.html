30 // Represents a filename path, e.g., /usr/local/bin/file.exe
31 // FIXME: Can we replace with std::filesystem?
32 class PathName final {
33  private:
34  static const char pathSeparators[2];
35  cstring str;
36 
37  const char *findLastSeparator() const;
38 
39  public:
40  static inline cstring separator() {
41 #ifdef _WIN32
42  return "\\"_cs;
43 #else
44  return "/"_cs;
45 #endif
46  }
47 
48  PathName(cstring str) : str(str) {} // NOLINT(runtime/explicit)
49  PathName(const char *str) : str(str) {} // NOLINT(runtime/explicit)
50  PathName(const std::string &str) : str(str) {} // NOLINT(runtime/explicit)
51  // get the file name extension. It starts at the last dot.
52  // e.g, exe
53  cstring getExtension() const;
54  // extract just the filename, including the extension
55  // e.g., file.exe
56  PathName getFilename() const;
57  // extract the filename without folder, excluding the extension
58  // e.g., file
59  cstring getBasename() const;
60  // extract the folder
61  // e.g., /usr/local/bin
62  PathName getFolder() const;
63  cstring toString() const { return str; }
64  bool isNullOrEmpty() const { return str.isNullOrEmpty(); }
65  bool operator==(const PathName &other) const { return str == other.str; }
66  bool operator!=(const PathName &other) const { return str != other.str; }
67  PathName join(cstring component) const;
68 
69  static PathName empty;
70 };
71 } // namespace Util
