111  template <MessageType type, int errorCode>
112  bool add(const char *name, bool forceReplace = false) {
113  static_assert(type != MessageType::Error ||
114  (errorCode >= ErrorType::ERR_MIN_BACKEND && errorCode <= ErrorType::ERR_MAX));
115  static_assert(type != MessageType::Warning || (errorCode >= ErrorType::WARN_MIN_BACKEND &&
116  errorCode <= ErrorType::WARN_MAX));
117  static_assert(type != MessageType::Info || (errorCode >= ErrorType::INFO_MIN_BACKEND &&
118  errorCode <= ErrorType::INFO_MAX));
119  static_assert(type != MessageType::None);
120  if (forceReplace) errorCatalog.erase(errorCode);
121  auto it = errorCatalog.emplace(errorCode, name);
122  return it.second;
123  }
124 
126  cstring getName(int errorCode) {
127  using namespace P4::literals;
128 
129  if (errorCatalog.count(errorCode)) return errorCatalog.at(errorCode);
130  return "--unknown--"_cs;
131  }

134  bool isError(std::string_view name) {
135  cstring lookup(name);
136  // Some diagnostics might be both errors and warning/info
137  // (e.g. "invalid" -> both ERR_INVALID and WARN_INVALID).
138  bool error = false;
139  for (const auto &pair : errorCatalog) {
140  if (pair.second == lookup) {
141  if (pair.first < ErrorType::LEGACY_ERROR || pair.first > ErrorType::ERR_MAX)
142  return false;
143  error = true;
144  }
145  }
146 
147  return error;
148  }
149 
150  private:
151  ErrorCatalog() {}

155 };
