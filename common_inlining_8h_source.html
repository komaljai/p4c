56 template <class Callable, class CallNode, class CallInfo>
57 class SimpleInlineWorkList : public IHasDbPrint {
58  public:
59  // Map caller -> statement -> callee
60  std::map<const Callable *, std::map<const CallNode *, const Callable *>> sites;
61  void add(CallInfo *info) {
62  CHECK_NULL(info);
63  LOG3(info);
64  sites[info->caller][info->call] = info->callee;
65  }
66  void dbprint(std::ostream &out) const {
67  for (auto t : sites) {
68  out << dbp(t.first);

70  out << std::endl << "\t" << dbp(c.first) << " => " << dbp(c.second);
71  }
72  }
73  }
74  bool empty() const { return sites.empty(); }

77 template <class Callable, class CallInfo, class InlineWorkList>
78 class SimpleInlineList {
79  std::vector<CallInfo *> toInline; // initial data
80  std::vector<CallInfo *> inlineOrder; // sorted in inlining order
81 
82  public:
83  // generate the inlining order
84  void analyze() {
85  // We only keep the call graph between objects of the same kind.
86  P4::CallGraph<const Callable *> cg("Call-graph");
87  for (auto c : toInline) cg.calls(c->caller, c->callee);
88 
89  // must inline from leaves up
90  std::vector<const Callable *> order;
91  cg.sort(order);
92  for (auto c : order) {
93  // This is quadratic, but hopefully the call graph is not too large
94  for (auto ci : toInline) {
95  if (ci->caller == c) inlineOrder.push_back(ci);
96  }
97  }
98 
99  std::reverse(inlineOrder.begin(), inlineOrder.end());
100  }
101 
102  size_t size() const { return toInline.size(); }
103 
105  InlineWorkList *next() {
106  if (inlineOrder.size() == 0) return nullptr;
107 
108  std::set<const Callable *> callers;
109  auto result = new InlineWorkList();
110 
111  // Find callables that can be inlined simultaneously.
112  // This traversal is in topological order starting from leaf callees.
113  // We stop at the first callable which calls one of the callables
114  // we have already selected.
115  while (!inlineOrder.empty()) {
116  auto last = inlineOrder.back();
117  if (callers.find(last->callee) != callers.end()) break;
118  inlineOrder.pop_back();
119  result->add(last);
120  callers.emplace(last->caller);
121  }
122  BUG_CHECK(!result->empty(), "Empty list of methods to inline");
123  return result;
124  }
125 
126  void add(CallInfo *aci) { toInline.push_back(aci); }
127 
128  void replace(const Callable *container, const Callable *replacement) {
129  LOG2("Substituting " << container << " with " << replacement);
130  for (auto e : inlineOrder) {
131  if (e->callee == container) e->callee = replacement;
132  if (e->caller == container) e->caller = replacement;
133  }
134  }
135 };

159 template <class InlineList, class InlineWorkList>
160 class InlineDriver : public Visitor {
161  InlineList *toInline;
162  AbstractInliner<InlineList, InlineWorkList> *inliner;
163 
164  public:
165  InlineDriver(InlineList *toInline, AbstractInliner<InlineList, InlineWorkList> *inliner)
166  : toInline(toInline), inliner(inliner) {
167  CHECK_NULL(toInline);
168  CHECK_NULL(inliner);
169  setName(("InlineDriver_"_cs + cstring(inliner->name())).c_str());
170  }
171  const IR::Node *apply_visitor(const IR::Node *program, const char * = 0) override {
172  LOG2("InlineDriver");
173  toInline->analyze();
174  LOG3("InlineList size " << toInline->size());
175  while (auto todo = toInline->next()) {
176  LOG2("Processing " << todo);
177  inliner->prepare(toInline, todo);
178  program = program->apply(*inliner);
179  if (::errorCount() > 0) break;
180 
181 #if DEBUG_INLINER
182  // debugging code; we don't have an easy way to dump the program here,
183  // since we are not between passes
184  ToP4 top4(&std::cout, false, nullptr);
185  program->apply(top4);
186 #endif
187  }
188  return program;
189  }
190 };
