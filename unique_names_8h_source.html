90  setName("FindSymbols");

93  cstring newName = refMap->newName(decl->getName().name.string_view());

96  void postorder(const IR::Declaration_Variable *decl) override { doDecl(decl); }
97  void postorder(const IR::Declaration_Constant *decl) override {
98  // Skip toplevel constants with names like __
99  // We assume that these do not clash and no new symbols with
100  // these names will be added.
101  if (decl->getName().name.startsWith("__") && getParent<IR::P4Program>()) return;
102  doDecl(decl);
103  }
104  void postorder(const IR::Declaration_Instance *decl) override {
105  if (!isTopLevel()) doDecl(decl);
106  }
107  void postorder(const IR::P4Table *decl) override { doDecl(decl); }
108  void postorder(const IR::P4Action *decl) override {
109  if (!isTopLevel()) doDecl(decl);
110  }
111  void postorder(const IR::P4ValueSet *decl) override {
112  if (!isTopLevel()) doDecl(decl);
113  }
