32 class DoStaticAssert : public Transform {
33  ReferenceMap *refMap;
34  TypeMap *typeMap;
35  bool removeStatement;
36 
37  // Cannot go static here as cstring is not constexpr
38  const cstring staticAssertMethodName = "static_assert"_cs;
39 
40  public:
41  DoStaticAssert(ReferenceMap *refMap, TypeMap *typeMap)
42  : refMap(refMap), typeMap(typeMap), removeStatement(false) {
43  CHECK_NULL(refMap);
44  CHECK_NULL(typeMap);
45  setName("DoStaticAssert");
46  }
47  const IR::Node *postorder(IR::MethodCallExpression *method) override {
48  MethodInstance *mi = MethodInstance::resolve(method, refMap, typeMap);
49  if (auto ef = mi->to<ExternFunction>()) {
50  if (ef->method->name == staticAssertMethodName) {
51  auto subst = ef->substitution;
52  auto params = subst.getParametersInOrder();
53  if (!params->moveNext()) {
54  ::warning(ErrorType::WARN_INVALID, "static_assert with no arguments: %1%",
55  method);
56  return method;
57  }
58  auto param = params->getCurrent();
59  CHECK_NULL(param);
60  auto arg = subst.lookup(param);
61  CHECK_NULL(arg);
62  if (auto bl = arg->expression->to<IR::BoolLiteral>()) {
63  if (!bl->value) {
64  cstring message = "static_assert failed"_cs;
65  if (params->moveNext()) {
66  param = params->getCurrent();
67  CHECK_NULL(param);
68  auto msg = subst.lookup(param);
69  CHECK_NULL(msg);
70  if (auto sl = msg->expression->to<IR::StringLiteral>()) {
71  message = sl->value;
72  }
73  }
74  ::error(ErrorType::ERR_EXPECTED, "%1%: %2%", method, message);
75  return method;
76  }
77  if (getContext()->node->is<IR::MethodCallStatement>()) {
78  removeStatement = true;
79  return method;
80  }
81  return new IR::BoolLiteral(method->srcInfo, true);
82  } else {
83  ::error(ErrorType::ERR_UNEXPECTED,
84  "Could not evaluate static_assert to a constant: %1%", arg);
85  return method;
86  }
87  }
88  }
89  return method;
90  }
91 
92  const IR::Node *postorder(IR::MethodCallStatement *statement) override {
93  if (removeStatement) {
94  removeStatement = false;
95  return nullptr;
96  }
97  return statement;
98  }
99 };
